//sort 연습
#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
// 선택 정렬 : 가장 작은 값을 앞으로 넣는 과정
void selectionSort(int *arr, int size){
    for(int i=0; i<size; i++){
        int num = arr[i];
        int index = i;
        for(int j = i+1; j<size ; j++){
            if(arr[j] < num){   //최소값 찾기
                num = arr[j];   //최소값의 index와 값 저장
                index = j;
            }
        }
        arr[index] = arr[i];
        arr[i] = num;
    }
}

void insertionSort(int n, vector<int> arr) {
    for(int i=1; i<n; i++){
        int num = arr[i];
        for(int k=i-1; i>0; k--){
            if(num <arr[k]){
                arr[k+1]= arr[k];
            }else{
                arr[k+1] = num;
                break;
            }
        }
        for(int temp : arr){
            cout << temp <<" ";
        }cout << "\n";
    }
}
