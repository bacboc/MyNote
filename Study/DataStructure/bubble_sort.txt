//CodeGround Note
// Bubble Sort: N-1 번 인접한 것 끼리 비교해서 큰것을 오른쪽으로
// 시간복잡도 N^2

#include<iostream>
using namespace std;
int d[5005];
void Swap(int &a, int &b){
    int temp = a;
    a=b;
    b=temp;
}
/*
void Swap(int *a, int *b){
    int temp = *a;
    *a=*b;
    *b=temp;
}
*/
int main() {
    int n;
    cin >> n;
    
    for(int i=0; i<n; i++){
        cin >> d[i];
    }
    
    //bubble
    for(int i=0; i<n-1; i++){
        for(int j=0; j<n-i-1;j++){
            if(d[j]>d[j+1]){
                Swap(d[j],d[j+1]);
            }
        }
    }
    for(int i=0; i<n; i++){
        cout << d[i] <<" ";
    }

    return 0;
}
