void Swap(int &a, int &b){
    int temp = a;
    a=b;
    b=temp;
}
void selection_sort(int* arr, int size){
    for(int i=0; i<size; i++){
        int minidx=i;
        for(int j=i+1;j<size; j++){
            if(arr[minidx]>arr[j]){
                minidx=j;
            }
        }
        Swap(arr[minidx],arr[i]);
    }
}
