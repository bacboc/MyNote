//조건문 없이 절대값
// 배울점: BIT 쉬프트 연산은 부호비트에 따라
// 부호비트가 1이면 1로 채워지고
// 부호비트가 0이면 0로 채워지고

#include<iostream>
using namespace std;

int Abs(int n){
    int temp = n>>31;
    cout << "temp : "<<temp << "\n";
    temp = (temp^n) + (temp&1);
    return temp;
}
int main(){
    int n;
    cin >> n;
    cout << Abs(n) << " ";
    return 0;
}
