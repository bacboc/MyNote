// Complete the sumXor function below.
long sumXor(long n) {
    long mod=0;
    while(n!=0){
        if(n%2 == 0){ // bit 0의 개수 구하기
            mod++;
        }
        n=n>>1;
    }
    return (long)1 << mod;
}

//Solution
long sumXor(long n)
{
    long ans=1;
    while(n)
    {
        long b=n&1;
        n=n>>1;
        if(!b) ans=ans*2;
    }
    return ans;
}
