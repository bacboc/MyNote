//백준 [이모티콘]
// BFS의 상태가 있다는 것 => visited잘 둬야한다.

#include<iostream>
#include<queue>
using namespace std;
int S;
struct node{
    int screen, board, depth;
    node(int s, int b, int d): screen(s), board(b), depth(d){}
    node(){}
    
};
bool visited[1001][1001];
int main(){
    cin >>S;
    queue<node> Q;
    node start(1,0,0);
    Q.push(start);
    while(!Q.empty()){
        node temp = Q.front();
        Q.pop();
        if(S == temp.screen){
            cout << temp.depth;
            break;
        }
        if(visited[temp.screen][temp.screen]==false){// 클립보드에 복사
            visited[temp.screen][temp.screen]=true;
            Q.push(node(temp.screen, temp.screen, temp.depth+1));
        }
        
        if(temp.screen+temp.board<1001&&visited[temp.screen+temp.board][temp.board]==false){
            // 화면에 붙여넣기
            visited[temp.screen+temp.board][temp.board]=true;
            Q.push(node(temp.screen+temp.board, temp.board, temp.depth+1));
        }
        
        if(temp.screen>0&&visited[temp.screen-1][temp.board]==false){
            // 화면에서 하나 삭제
            visited[temp.screen-1][temp.board]=true;
            Q.push(node(temp.screen-1, temp.board, temp.depth+1));
        }
     
    }
    return 0;
}
