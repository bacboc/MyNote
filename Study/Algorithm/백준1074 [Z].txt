#include<iostream>
#include<cmath>
#include<string>
using namespace std;
int N;
long long R, C;
long long sol=0;

void recursion(int n){
    if(R==0 && C==0){
        //sol
    }else if( R==0 && C==1){
        sol += 1;
    }else if( R==1 && C==0){
        sol += 2;
    }else if( R==1 && C==1){
        sol += 3;
    }else{
        long long s= pow(2,n-1);
        //type
        if(R < s && C <s){ //0 영역
            // 그대로 n 만 줄여서 다시 계산
        }else if(R < s && C >= s){ // 1 영역
            sol += (long long)pow(4,n-1);
            C -= s;
        }else if(R >= s && C < s){ // 2
            sol += (long long)pow(4,n-1)*2;
            R -= s;
        }else if(R >=s && C >= s){ // 3
            sol += (long long)pow(4,n-1)*3;
            C -= s;
            R -= s;
        }
        recursion(n-1);
    }
}
int main(){
    scanf("%d %lld %lld", &N, &R, &C);
    if(R==0&& C==0){
        cout << 0;
        return 0;
    }
    recursion(N);
    printf("%lld",sol);
    return 0;
}
