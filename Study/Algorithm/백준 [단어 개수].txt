/* 백준 단어 개수
 
    %[^\n]s 이거 아주 꿀 ==> \n 를 만날때 까지 문자열 쭉 저장한다는 의미 == gets 
 
 */

#include<stdio.h>

int main(){
    char str[1000005];
    scanf("%[^\n]s", str);
    //gets(str);
    int k=0;
    int sol=0;
    int flag = 1;
    while(str[k]==' '){
        k++;
    }
    while(str[k]!='\0'){
        if(str[k]==' '){
            flag=1;
        }else{
            if(flag){
                sol++;
                flag=0;
            }
        }
        k++;
    }
    
    printf("%d",sol);
    
    return 0;
}
